TransitStage.cpp の 38 行で定義されています。

参照先 _next_main_stage, _pMessage, _TRACE_, processBehaviorProgPlaying(), MyStg2nd::STAGESCENE_PROG_BEGIN, MyStg2nd::STAGESCENE_PROG_END, MyStg2nd::STAGESCENE_PROG_INIT, と MyStg2nd::STAGESCENE_PROG_PLAYING.

                                   {
    StageScene::processBehavior();

    switch (_pProg->get()) {
        case STAGESCENE_PROG_INIT: {
            _TRACE_("TransitStage::processBehavior() Prog(=STAGESCENE_PROG_INIT) is Just Changed");
            _pProg->change(STAGESCENE_PROG_BEGIN);
            break;
        }
        case STAGESCENE_PROG_BEGIN: {
            if (_pProg->isJustChanged()) {
                _TRACE_("TransitStage::processBehavior() Prog(=STAGESCENE_PROG_BEGIN) is Just Changed");
            }

            if (_pProg->getFrameInProgress() == 180) { //ステージ１開始！
                _pProg->change(STAGESCENE_PROG_PLAYING);
            }
            break;
        }
        case STAGESCENE_PROG_PLAYING: {
            if (_pProg->isJustChanged()) {
                _TRACE_("TransitStage::processBehavior() Prog(=STAGESCENE_PROG_PLAYING) is Just Changed");
            }

            if (_pProg->getFrameInProgress() == 60) { //ステージ開始！
                _pMessage->update("SELECT NEXT STAGE!");
//                _pMessage->inactivateDelay(240);
            }

            processBehaviorProgPlaying();

            break;
        }

        case STAGESCENE_PROG_END: {
            if (_pProg->isJustChanged()) {
                _TRACE_("TransitStage::processBehavior() Prog(=STAGESCENE_PROG_END) is Just Changed");
                throwEventToUpperTree(EVENT_PREPARE_NEXT_STAGE, (void*)_next_main_stage); //次ステージ準備へ
            }

            if (_pProg->getFrameInProgress() == 120) {
                _pMessage->update("GOOD LUCK!");
            }

            if (_pProg->getFrameInProgress() == 300) {
                _TRACE_("TransitStage::processBehavior() Prog(=STAGESCENE_PROG_END) throwEventToUpperTree(EVENT_TRANSIT_WAS_END)");
                throwEventToUpperTree(EVENT_TRANSIT_WAS_END);
            }

            break;
        }
        default:
            break;

    }

}
