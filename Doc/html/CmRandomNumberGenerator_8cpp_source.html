00119 void CmRandomNumberGenerator::init_genrand(unsigned long s) {

00134 unsigned long CmRandomNumberGenerator::genrand_int32(void) {
00135     unsigned long y;
00136     static unsigned long mag01[2] = {0x0UL, MATRIX_A};
00137     /* mag01[x] = x * MATRIX_A  for x=0,1 */
00138 
00139     if (mti >= __N__) {
00140         /* generate __N__ words at one time */
00141         int kk;
00142 
00143         if (mti == __N__ + 1) /* if init_genrand() has not been called, */
00144         init_genrand(5489UL); /* a default initial seed is used */
00145 
00146         for (kk = 0; kk < __N__ - __M__; kk++) {
00147             y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
00148             mt[kk] = mt[kk + __M__] ^ (y >> 1) ^ mag01[y & 0x1UL];
00149         }
00150 
00151         for (; kk < __N__ - 1; kk++) {
00152             y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
00153             mt[kk] = mt[kk + (__M__ - __N__)] ^ (y >> 1) ^ mag01[y & 0x1UL];
00154         }
00155 
00156         y = (mt[__N__ - 1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
00157         mt[__N__ - 1] = mt[__M__ - 1] ^ (y >> 1) ^ mag01[y & 0x1UL];
00158 
00159         mti = 0;
00160     }
00161 
00162     y = mt[mti++];
00163 
00164     /* Tempering */
00165     y ^= (y >> 11);
00166     y ^= (y << 7) & 0x9d2c5680UL;
00167     y ^= (y << 15) & 0xefc60000UL;
00168     y ^= (y >> 18);
00169
